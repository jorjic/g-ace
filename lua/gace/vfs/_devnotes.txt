Note: this is not up to date documentation, but notes made during development of vfs

class Node
    Events: capabilitiesChanged

    static Bit CAPABILITY_READ = 1 << 0
    static Bit CAPABILITY_WRITE = 1 << 1
    static Bit CAPABILITY_REALFILE = 1 << 2 --is a file/folder on a real filesystem

    abstract String getName()

    abstract Bitfield32 getCapabilities()
    Boolean hasCapability()

    Node getParent()
    String path()

    abstract String realPath() -- if hasCapability(CAPABILITY_REALFILE)

class File extends Node
    Events: deleted

    static Bit CAPABILITY_STAT = 1 << 8 -- lastmod and size

    abstract Number getLastModified() -- unix timestamp
    abstract Number getSize()

    abstract Promise read([options])
    abstract Promise write(data, [options])

class Folder extends Node

    abstract Promise listEntries([options])
    abstract Promise createChildNode(name, type, [options])
